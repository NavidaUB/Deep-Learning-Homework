class Server:
    
    def __init__(self,m,d,lr):
        self.m = m
        self.d = d
        self.w = numpy.zeros((d,1))
        self.lr=lr
        
    def broadcast (self):
        return self.w ,self.lr
    
    def aggregate(self,ps):
        P=numpy.zeros((self.d,1))
        for k in range(self.m):
            P +=ps[k]
        P=P/self.m
        self.w -= self.lr*P
    
    def lr_update(self,update_ratio):
        self.lr *= update_ratio

class Worker:
    def __init__(self,x,y):
        self.x=x
        self.y=y
        self.s=x.shape[0]
        self.d=x.shape[1]
        self.w=numpy.zeros((self.d,1))
        self.lr=0
        
    def set_param(self,w,lr):
        self.w=w
        self.lr=lr
    
    def loss(self,lam):
        yx=numpy.multiply(self.y,self.x)
        yxw=numpy.dot(yx,self.w)
        vec1=numpy.exp(-yxw)
        vec2=numpy.log(1+vec1)
        reg = lam/2 * numpy.sum(self.w*self.w)
        return numpy.sum(vec2)/self.s + 4*reg
    
    def gradient(self,lam,q,b):        
        w_checkpoint = self.w   
        for qq in range(q):                  
            if b=="SGD":  
                for j in range(self.s):
                    temp_y=self.y[j,:]
                    temp_x=self.x[j,:]
                    yx=numpy.multiply(temp_y,temp_x)
                    yxw=numpy.dot(yx,self.w)
                    vec1=numpy.exp(yxw)
                    vec2=numpy.divide(yx,1+vec1)
                    g=-numpy.sum(vec2,axis=0).reshape(self.d,1) + lam*self.w
                    self.w -= self.lr*g                                   
            else:
                temp_y=self.y
                temp_x=self.x
                yx=numpy.multiply(temp_y,temp_x)
                yxw=numpy.dot(yx,self.w)
                vec1=numpy.exp(yxw)
                vec2=numpy.divide(yx,1+vec1)
                g=-numpy.sum(vec2,axis=0).reshape(self.d,1)/self.s + lam*self.w
                self.w -= self.lr*g
        p= w_checkpoint - self.w
        return p

import math

def create_network(m,x,y,lr):
    n,d=x.shape
    s=math.floor(n/m)
    server=Server(m,d,lr)
    workers=[]    
    for i in range(m):
        indices=list(range(i*s,(i+1)*s))
        worker=Worker(x[indices,:],y[indices,:])
        workers.append(worker)
    return server,workers

    for t in range(max_epoch):
        w,lr=server.broadcast()
        for i in range(m):
            workers[i].set_param(w,lr)        
        ps=[]
        loss=0
        for i in range(m):
            p=workers[i].gradient(lam,q,b)
            ps.append(p)
            loss+=workers[i].loss(lam)   
        final_losses[t]=loss/m      
        server.aggregate(ps)
        server.lr_update(update_ratio)
    return final_losses
